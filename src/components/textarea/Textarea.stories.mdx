<Canvas withSource={SourceState.OPEN}>
  <Story
    name="With icon"
    args={{
      iconLeft: icons[0],
      theme: 'dark',
      textAlign: 'left',
      error: false,
      disabled: false,
      readOnly: false,
      resizable: false,
    }}
    argTypes={{
      iconLeft: {
        options: icons,
        control: { type: 'select' },
      },
      iconRight: {
        options: icons,
        control: { type: 'select' },
      },
      theme: {
        table: { category: 'Input args' },
      },
      textAlign: {
        table: { category: 'Input args' },
      },
      error: {
        table: { category: 'Input args' },
      },
      disabled: {
        table: { category: 'Input args' },
      },
      readOnly: {
        table: { category: 'Input args' },
      },
      resizable: {
        table: { category: 'Input args' },
      },
    }}
  >
    {({ iconLeft, iconRight, theme, textAlign, error, disabled, readOnly, resizable }) => (
      <lp-textarea
        key={`${iconLeft}-${iconRight}`}
        theme={theme === 'dark' ? undefined : theme}
        textAlign={textAlign === 'left' ? undefined : textAlign}
        error={error === false ? undefined : error}
        disabled={disabled === false ? undefined : disabled}
        readOnly={readOnly === false ? undefined : readOnly}
        resizable={resizable === false ? undefined : resizable}
      >
        {!!iconLeft && createElement(`lp-${iconLeft}-icon`, { slot: 'icon-left' })}
        {!!iconRight && createElement(`lp-${iconRight}-icon`, { slot: 'icon-right' })}
      </lp-textarea>
    )}
  </Story>
</Canvas>
