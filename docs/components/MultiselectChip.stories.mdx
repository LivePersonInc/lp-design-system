## Basic

export const MultiselectChipTemplate = props => {
  const multiselectChipElRef = useRef(null);
  useEffect(() => {
    multiselectChipElRef.current.chips = Array(5).fill(null).map((item, index) => ({ label: `Chip ${++index}` }));
  }, []);
  return <lp-multiselect-chip ref={multiselectChipElRef} {...props} />;
};
