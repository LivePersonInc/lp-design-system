# Button

<ArgsTable of={Button} />

`label` value can be provided by `label` attribute or by `slot` tag with the attribute `slot="label"`

Example:
```
<lp-button label="Button"></lp-button>
```
or
```
<lp-button>
  <slot slot="label">Button</slot>
</lp-button>
```

## With icon

<Canvas withSource={SourceState.OPEN}>
  <Story
    name="With icon"
    args={{
      iconLeft: icons[0],
      theme: 'dark',
      variant: 'primary',
      size: 'medium',
      label: 'With icon',
      disabled: false,
    }}
    argTypes={{
      iconLeft: {
        options: icons,
        control: { type: 'select' },
      },
      iconRight: {
        options: icons,
        control: { type: 'select' },
      },
      theme: {
        table: { category: 'Button args' },
      },
      variant: {
        table: { category: 'Button args' },
      },
      size: {
        table: { category: 'Button args' },
      },
      label: {
        table: { category: 'Button args' },
      },
      disabled: {
        table: { category: 'Button args' },
      },
    }}
  >
    {({ iconLeft, iconRight, theme, variant, size, label, disabled }) => (
      <lp-button
        key={`${iconLeft}-${iconRight}`}
        theme={theme === 'dark' ? undefined : theme}
        variant={variant === 'primary' ? undefined : variant}
        size={size === 'medium' ? undefined : size}
        label={label}
        disabled={disabled === false ? undefined : disabled}
      >
        {!!iconLeft && createElement(`lp-${iconLeft}-icon`, { slot: 'icon-left' })}
        {!!iconRight && createElement(`lp-${iconRight}-icon`, { slot: 'icon-right' })}
      </lp-button>
    )}
  </Story>
</Canvas>
