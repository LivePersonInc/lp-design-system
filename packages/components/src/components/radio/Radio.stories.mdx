export const Template = ({ name, selected, elements, ...props }) => (
  <lp-radio-group
    key={`${elements.toString()}-${selected}`}
    name={name}
    selected={selected}
    {...getPropsByDefaultProps(radioDefaultProps, props)}
  >
    {elements.map(element => <lp-radio key={element} label={element} value={element} />)}
  </lp-radio-group>
)

# Radio

Radio component always has to work with RadioGroup component. For this, you don't need to import two modules.
RadioGroup and Radio components place in one `radio` module

```javascript
import '@liveperson-design-system/components/radio';
```
